1#
  
2# Any user who can edit a {Page}[link:classes/Page.html] can also upload attachments to that page.
  
3# (Note: in this system a Page is the CMS pages that represent curriculum materials. A user can not
  
4# upload an attachment to a team or a person page through this code.)
  
5#
  
6# The attachments are stored on Amazon S3. S3 allows us to revision all documents. All old versions
  
7# are still stored on the filesystem even though the user can only actively access the latest one.
  
8#
  
 9# When uploading a file, the user is asked for a "readable name" for the file: a label for the link
  
10# to the file.
  
11#
  
12# After uploading a file, the file's "readable name" can be updated or the file can be replaced.
  
13# If the user replaces the an existing file (e.g. 'ppm_week1.ppt') with a different file name
  
14# (e.g. 'ppm_week1_ts.ppt') then a warning appears. Any user who tries to access the original file
  
15# name will not be able to do so. Under normal workflow, this isn't a big deal. It is possible
  
16# for the user to create a link on the page to the old file and system will not automatically
  
17# update it.
  
18#
  
19# The system records who uploads and replaces the file and when this happens, and versions this information.
  
20#
  
21
  
22class PageAttachment < ActiveRecord::Base
  
23  belongs_to :page
  
24  belongs_to :user
  
25
  
26  validates_presence_of :readable_name, :user_id, :page_id
  
27
  
28  has_attached_file :page_attachment,
  
29                    :storage => :s3,
  
30                    :bucket => ENV['WHITEBOARD_S3_BUCKET'],
  
31                    :s3_credentials => {:access_key_id => ENV['WHITEBOARD_S3_KEY'],
  
32                                        :secret_access_key => ENV['WHITEBOARD_S3_SECRET']},
  
33                    :path => "page_attachments/:page_id/:random_hash/:id/:filename"
  
34
  
35  versioned
  
36

38  def self.reposition(ids)
  
39    update_all(["position = STRPOS(?, ','||id||',')", ",#{ids.join(',')},"], {:id => ids})
  
40  end
  
41
  
42end
