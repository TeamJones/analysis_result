1require 'rubygems'
  
2require 'aws-sdk'
  
3require 'fileutils'
  
4
  
5desc "Create nondigest versions of all ckeditor digest assets"
  • Score of 61.34 » flog
 
6task "assets:precompile" => :environment do
  
7  AWS.config(
  
8      :access_key_id =>  ENV['WHITEBOARD_S3_KEY'],
  
 9      :secret_access_key => ENV['WHITEBOARD_S3_SECRET']
  
10  )
  
11  # Get an instance of the S3 interface.
  
12  s3 = AWS::S3.new
  
13
  
14  fingerprint = /\-([0-9a-f]{32})\./
  
15  for file in Dir["public/assets/ckeditor/**/*"]
  
16    # Skip file unless it has a fingerprint
  
17    next unless file =~ fingerprint
  
18
  
19    # Get filename of this file without the digest
  
20    # (example) public/assets/ckeditor/config.js
  
21    nondigest = file.sub fingerprint, '.'
  
22
  
23    # Create a filename relative to public/assets
  
24    # (example) public/assets/ckeditor/config.js => ckeditor/config.js
  
25    filename = nondigest.sub 'public/assets/', ''
  
26    filename = filename.sub /.gz$/, ''          # Remove .gz for correct asset checking
  
27
  
28    # Fetch the latest digest for this file from assets
  
29    latest_digest = Rails.application.assets.find_asset(filename).digest
  
30
  
31    # Debug information
  
32    puts '---- ' + file + ' ----'
  
33
  
34    # Compare digest of this file to latest digest
  
35    # [1] is the enclosed capture in the fingerprint regex above
  
36    this_digest = file.match(fingerprint)[1]
  
37    if (this_digest == latest_digest)
  
38      # This file's digest matches latest digest, copy
  
39      puts 'Matching digest, copying ' + file
  
40      FileUtils.cp file, nondigest, verbose: true
  
41      # Upload a file.
  
42      key = File.basename(nondigest)
  
43      s3.buckets[ENV['WHITEBOARD_S3_ASSET_BUCKET']].objects[key].write(:file => nondigest)
  
44      puts "Uploading file #{nondigest} to bucket #{ENV['WHITEBOARD_S3_ASSET_BUCKET']}."
  
45    else
  
46      # This file's digest doesn't match latest digest, ignore
  
47      puts 'Latest digest: ' + latest_digest
  
48      puts 'This digest:   ' + this_digest
  
49      puts 'Non-matching digest, not copying ' + file
  
50    end
  
51
  
52    # Debug information
  
53    puts '---- end ----'
  
54  end
  
55end
