11  def self.default_search_results(user)
  
12    # Grab all records from the people_search_defaults table
  
13    search_defaults = PeopleSearchDefault.all
  
14
  
15    # Save "all" records for student_staff_group into results and reject from search_defaults list to move forward.
  
16    results = search_defaults.find_all { |t| t.student_staff_group == 'All' }
  
17    search_defaults.reject!  { |t| t.student_staff_group == 'All' }
  
18    if(user.is_student)
  
19
  
20      # Filter to only students
  
21      search_defaults = search_defaults.find_all { |t| t.student_staff_group == 'Student' }
  
22
  
23      # Save "all" records for program_group into results and reject from search_defaults list to move forward.
  
24      search_defaults.find_all { |t| t.program_group == 'All' }.each { |x| results.push(x) }
  
25      search_defaults.reject!  { |t| t.program_group == 'All' }
  
26
  
27      # Filter to only students in same master program as user
  
28      search_defaults = search_defaults.find_all { |t| t.program_group == user.masters_program }
  
29
  
30      # Save "all" records for track_group into results and reject from search_defaults list to move forward.
  
31      search_defaults.find_all { |t| t.track_group == 'All' }.each { |x| results.push(x) }
  
32      search_defaults.reject!  { |t| t.track_group == 'All' }
  
33
  
34      # Filter to only students in same master's track as user
  
35      search_defaults = search_defaults.find_all { |t| t.track_group == user.masters_track || (user.masters_program == 'PhD' && t.track_group == nil) }.each { |x| results.push(x) }
  
36
  
37    else
  
38      # Filter to only staff and save
  
39      search_defaults = search_defaults.find_all { |t| t.student_staff_group == 'Staff' }
  
40      search_defaults.each { |x| results.push(x) }
  
41    end
  
42    # remove user if he would see himself and return.
  
43    results.reject{ |t| t.user_id == user.id }
  
44  end
