21    function enable_tabs(checkbox) {
  
22        if (checkbox.prop('checked')) {
  
23            $tabs = $('#tabs').tabs();
  
24            $tabs.tabs('option', 'disabled', []); //enables the tabs
  
25            $tabs.tabs('option', 'selected', parseInt($.cookie('selected_tab'), 10));
  
26            $('.ui-tabs-nav').css('display', 'block');
  
27            $('#email_header').css('display', 'block');
  
28            $('.page_or_task').html("task");
  
29        }
  
30        else {
  
31            $tabs = $('#tabs').tabs();
  
32            $tabs.tabs('option', 'selected', 0);
  
33            $tabs.tabs('option', 'disabled', [1, 2, 3]);
  
34            $('.ui-tabs-nav').css('display', 'none');
  
35            $('#email_header').css('display', 'none');
  
36            $('.page_or_task').html("page");
  
37        }
  
38    }

40    function change_detected() {
  
41        needs_to_be_saved = true;
  
42        time_since_last_key_down = current_time();
  
43    }
  
44
  
45    function enableAutoSave() {
  
46        var postPath = $('#edit-form').attr('action');
  
47
  
48        ['page_tab_one_contents', 'page_tab_two_contents', 'page_tab_three_contents', 'page_faculty_notes'].forEach(function (editor) {
  
49            CKEDITOR.instances[editor].on('key', function () {
  
50                change_detected();
  
51            });
  
52        })
  
53
  
54        setInterval(function () {
  
55            if (needs_to_be_saved) {
  
56                var instance;
  
57                for (instance in CKEDITOR.instances) {
  
58                    CKEDITOR.instances[instance].updateElement();
  
59                }
  
60
  
61                $.ajax({
  
62                    type: 'post',
  
63                    dataType: 'json',
  
64                    data: $('#edit-form').serialize(),
  
65                    timeout: 8000,
  
66                    url: postPath,
  
67                    success: function (data) {
  
68                        if (data.code == 'success') {
  
69                            // Update the form with the (possibly) new path
  
70                            postPath = data['new_post_path'];
  
71                            $('#edit-form').attr('action', postPath);
  
72                            hideAjaxError();
  
73                            needs_to_be_saved = false;
  
74                        } else {
  
75                            showAjaxError(data['message']);
  
76                        }
  
77                    },
  
78                    error: function (request, status, message) {
  
79                        // ignore timeout error
  
80                        if ("timeout" != status) {
  
81                            if(message) {
  
82                                showAjaxError(message);
  
83                            } else {
  
84                                showAjaxError("Unable to auto-save. Perhaps the server is down.");
  
85                            }
  
86                        }
  
87                    },
  
88                    complete: function (data) {
  
89                        needs_to_be_saved = false;
  
90                    }
  
91                });
  
92            }
  
93        }, 10000);
  
94    }
  
95
  
96    function current_time() {
  
97        return new Date().getTime();
  
98    }
  
 99
  
100    function enableFiveMinuteTimeout() {
  
101        setInterval(function () {
  
102            // console.log("current_time  " + current_time() + "     time_since_last_key_down  " + time_since_last_key_down)
  
103            if((current_time() - time_since_last_key_down) >= (5 * 60 * 1000)) {
  
104                $("#timeout-flag").val("true");
  
105                // Disable popup warning user that they are leaving the page.
  
106                window.onbeforeunload = null;
  
107                $("#edit-form").submit();
  
108            }
  
109        }, 10000);
  
110    }

127          });
  
128
  
129          $('.auto-save-on-keydown').keydown(function () {

132        }

135            enableAutoSave();
  
136            alertify.log("Auto save is enabled. We'll save your changes every 10 seconds");
  
137        }

141        }

145        $.cookie('selected_tab', null);   //clear cookie

152              dialogDefinition.removeContents('Upload');
  
153              dialogDefinition.removeContents('upload');
  
154            }
  
155        });
