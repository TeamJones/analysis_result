4  task(:cleanup_user_bio_data => :environment) do
  
5      # we are keeping the biography column in the database
  
6      # we are relegating the user_text column in the database
  
7      users_that_need_scrubbing = User.where("users.biography != '' OR users.biography IS NOT NULL OR users.user_text != '' OR users.user_text IS NOT NULL ")
  
8                                      # .where(:twiki_name => "KaushikGopal")
  
 9          users_that_need_scrubbing.each do |u|
  
10              # clean up the data in biography
  
11              # remove boing boing text
  
12              u.biography = nil if u.biography.starts_with? "<p>I was raised by sheepherders on the hills of BoingBoing"
  
13
  
14              # clean up the data in user_text
  
15              # remove "<h2>About Me</h2>"
  
16              u.user_text.gsub!("<h2>About Me</h2>", "") unless u.user_text.blank?
  
17              u.user_text.gsub!("<p>I'd like to accomplish the following three goals (professional or personal) by the time I graduate:<ol><li>Goal 1</li><li>Goal 2</li><li>Goal 3</li></ol></p>", "") unless u.user_text.blank?
  
18              # <h2>About Me</h2><p>I'd like to accomplish the following three goals (professional or personal) by the time I graduate:<ol><li>Goal 1</li><li>Goal 2</li><li>Goal 3</li></ol></p>
  
19
  
20              # copy user_text data into biography
  
21              if u.biography.blank?
  
22                  u.biography = u.user_text unless u.user_text.blank?
  
23              else
  
24                  u.biography << u.user_text unless u.user_text.blank?
  
25              end
  
26
  
27              # clear text from user_text field
  
28              # u.user_text = nil
  
29              u.save!
  
30          end
  
31    end
  
32end
