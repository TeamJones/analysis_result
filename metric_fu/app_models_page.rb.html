83    options_hash = {:title => self.title, :type => "page"}
  
84    if self.course
  
85      options_hash.merge!({:title => self.title + " (" + self.course.name + ")"})
  
86    end

88    begin
  
89      index.document(self.id.to_s).add(options_hash.merge!({:text => self.tab_one_contents.gsub(/<\/?[^>]*>/, ""), :url => "pages/" + self.url}))
  
90      if self.is_task?
  
91        index.document(self.id.to_s + "-tabs-1").add(options_hash.merge!({:text => self.tab_two_contents.gsub(/<\/?[^>]*>/, ""), :url => "pages/" + self.url + "?tab=tabs-2"}))
  
92        index.document(self.id.to_s + "-tabs-2").add(options_hash.merge!({:text => self.tab_three_contents.gsub(/<\/?[^>]*>/, ""), :url => "pages/" + self.url + "?tab=tabs-3"}))
  
93      end
