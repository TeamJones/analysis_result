116<script>
  
117    $(document).ready(function () {
  
118        var $course_tools_view = $('#course_tools_view');
  
119
  
120        set_course_tools_view($course_tools_view, $course_tools_view.attr('data-view-state'));
  
121        $course_tools_view.click( function() {
  
122            old_state = $(this).attr('data-view-state');
  
123            if(old_state == "links"){
  
124              new_state = "icons";
  
125            }
  
126            else {
  
127              new_state = "links";
  
128            }
  
129            set_course_tools_view ($course_tools_view, new_state)
  
130            $.ajax({
  
131                type:'post',
  
132                data:"model=User&id=<%=current_user.id%>&attribute=course_tools_view&value=" + new_state,
  
133                dataType:'script',
  
134                url:'/generic_ajax/update_model_with_value'});
  
135        });
  
136    });
  
137
  
138    function set_course_tools_view ($ctv_obj, state) {
  
139      $ctv_obj.attr('data-view-state', state);
  
140      var $item_wrapper = $(".item-wrapper");
  
141      if (state == "links") {
  
142        $item_wrapper.addClass('links');
  
143        $ctv_obj.html("Show as icons");
  
144      }else{
  
145        $item_wrapper.removeClass('links');
  
146        $ctv_obj.html("Show as links");
  
147      }
  
148    }
  
149
  
150</script>
