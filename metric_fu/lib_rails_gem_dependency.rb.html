175      @loaded ||= begin
  
176        if vendor_rails?
  
177          true
  
178        elsif specification.nil?
  
179          false
  
180        else
  
181          # check if the gem is loaded by inspecting $"
  
182          # specification.files lists all the files contained in the gem
  
183          gem_files = specification.files
  
184          # select only the files contained in require_paths - typically in bin and lib
  
185          require_paths_regexp = Regexp.new("^(#{specification.require_paths*'|'})/")
  
186          gem_lib_files = gem_files.select { |f| require_paths_regexp.match(f) }
  
187          # chop the leading directory off - a typical file might be in
  
188          # lib/gem_name/file_name.rb, but it will be 'require'd as gem_name/file_name.rb

191          # have been loaded
